# 在vue项目中使用el-upload上传文件

# vuejs使用el-upload上传文件

我心里有片沙漠

今天介绍下如何在vue.js项目中使用饿了么中el-upload组件进行文件上传的操作。其中limit参数可以设置允许最大上传数量，http-request可以覆盖默认的上传行为，可以自定义上传的实现，同时需要把auto-upload参数的值设为false。

<el-upload
  :limit="1"
  class="upload-demo"
  ref="upload"
  action="xxx"
  :file-list="fileList"
  accept=".doc,.docx,application/msword,.pdf,"
  :http-request="uploadFile"
  :auto-upload="false"
  :before-upload="beforeUploadFunc">
  <el-button slot="trigger" type="primary" 
     style="width:103px;height:38px;">选取文件</el-button>
</el-upload>
<el-button type="primary" @click="submitToUpload" class="w100">确 定上传</el-button>

###Tips

可以设置`accept'参数的值来对上传文件的类型进行限制，如果选择的不是你在accept参数中设置的文件类型则会提示错误信息。 ###判断是否选择文件 其中submitToUpload()就是在点击上传按钮之前要执行的方法，可以对文件表单进行验证。

/*点击上传按钮时的执行方法*/
submitToUpload() {
  let list = document.getElementsByClassName('el-upload-list__item is-ready')
  if(list.length == 0){
    return this.$message({
      type:'warning',
      message:"请选择需要上传的文件！"
    })
}
  this.$refs.upload.submit();
 },

###判断文件的类型 下面的方法是在文件选择完成后，开始上传之前执行的方法，在这里可以根据自己需求来根据文件类型提示消息。

/*上传前的判断*/
beforeUploadFunc(file){
  const extension = file.name.split('.')[1] === 'docx';
  const extension2 = file.name.split('.')[1] === 'doc';
  const extension3 = file.name.split('.')[1] === 'pdf';
  const isLt2M = file.size / 1024 / 1024 < 2;
  if (!extension && !extension2 && !extension3) {
    this.$message.error('上传模板只能是 doc、docx、pdf格式!')
  }
  if (!isLt2M) {
    this.$message.error('上传文件大小不能超过 2MB!')
  }
  return extension || extension2 || extension3 && isLt2M
},

###上传文件 下面就是执行操作的方法了,使用了formData进行文件的传递，如果你需要加额外的参数，可以用使用form.append('参数名':'参数值')。

/*上传操作*/
uploadSFile(params){
  var fileObj = params.file;
/*FormData 对象*/
  var form = new FormData();
/*文件对象和额外参数*/
  form.append("file", fileObj);
  form.append("参数名", 参数值);
  this.$ajax.post('/train/tch/uploadCourseResource',form).then(res => {
/*上传成功执行的回调*/
    this.$message.success('上传成功');
    this.fileList =[];
  }).catch((error) => {
  /*提示错误信息*/
    this.$message({showClose: true,message: error,type: 'error'});
  });
